let zoom=1;const canvas=document.getElementById('canvas'),viewport=document.getElementById('viewport'),statusBox=document.getElementById('status'),updateBtn=document.getElementById('updateBtn'),fitBtn=document.getElementById('fitBtn'),thirdsToggle=document.getElementById('thirdsToggle');function showStatus(m,e=false){statusBox.textContent=m;statusBox.className='status show'+(e?' error':'');clearTimeout(showStatus.t);showStatus.t=setTimeout(()=>statusBox.className='status',7000)}function applyZoom(){canvas.style.transform=`scale(${zoom})`;canvas.style.width=`${1800*zoom}px`;canvas.style.height=`${1010*zoom}px`}function setZoom(v){zoom=v;applyZoom()}function fitWidth(){zoom=Math.max(.35,Math.min(1.5,viewport.clientWidth/1800));applyZoom()}async function loadBracket(){const r=await fetch('/api/render');const d=await r.json();canvas.innerHTML=d.svg;thirdsToggle.checked=d.include_thirds;applyZoom()}async function updatePromiedos(){showStatus('Actualizando Promiedos...');try{const r=await fetch('/api/update_promiedos',{method:'POST'});const d=await r.json();if(!d.ok)throw new Error(d.error||'Error al actualizar Promiedos');await loadBracket();showStatus(d.message||'Actualizado desde Promiedos.')}catch(e){showStatus(`Error: ${e.message}`,true)}}async function setThirds(en){await fetch('/api/settings',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({include_thirds:en})});await loadBracket()}document.querySelectorAll('[data-zoom]').forEach(b=>b.addEventListener('click',()=>setZoom(Number(b.dataset.zoom))));updateBtn.addEventListener('click',updatePromiedos);fitBtn.addEventListener('click',fitWidth);thirdsToggle.addEventListener('change',()=>setThirds(thirdsToggle.checked));window.addEventListener('resize',applyZoom);fitWidth();loadBracket();
